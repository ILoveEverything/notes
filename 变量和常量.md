# 变量和常量

## 1.单独声明变量

```go
var name string //默认对应""(空值)
var age int     //默认对应0
var isOK bool   //默认对应False
```

## 2.单独申明常量

```go
const pi = 3.1415926
```

## 3.批量声明变量

```go
var (
	name string //默认对应""(空值)
	age  int    //默认对应0
	isOK bool   //默认对应False
)
```

## 4.批量声明常量

```go
const (
	statusOk = 200
	notFound = 404
)
const ( //当s2与s3不赋值时，s2和s3的值和s1的值一致
	s1 = 10
	s2 //s2=10
	s3 //s3=10
)

//iota：在const关键字内出现时将被重置为0。const中没新增一行常量声明将使iota计数一次。（使用iota能简化定义，在定义枚举时很有用）
const (
	a1 = iota //a1=0
	a2        //a2=1
	a3        //a3=2
)

//定义数量级
const (
	_ = iota
	//表示1向左移动（10×iota）位的二进制数转换为十进制数
	KB = 1 << (10 * iota) //二进制10000000000=十进制1024
	MB = 1 << (10 * iota) //二进制100000000000000000000=十进制1048576
	GB = 1 << (10 * iota) //二进制1000000000000000000000000000000=十进制1073741824
	TB = 1 << (10 * iota) //二进制10000000000000000000000000000000000000000=十进制1.099511628×10¹²
	PB = 1 << (10 * iota) //二进制100000000000000000000000000000000000000000000000000=十进制1.125899907×10¹⁵
)
```
